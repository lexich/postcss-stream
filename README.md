Add support to postcss stream api. 
```js
const postcssStream = require('postcss-stream');

const stream1 = postcssStream.createStream({
   // select only nodes with type="decl" prop="color" value="red"
   query: {
        decl: {
            prop: "color",
            value: "red"
        }
   },
   fn(decl) {
     decl.value = "green";
   }
});

postcssStream([
  stream, ...
]);

```
